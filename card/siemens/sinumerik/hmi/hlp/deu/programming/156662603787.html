LOOP 
   IF NOT $P_SEARCH; IF kein Satzsuchlauf
      G1 G90 X0 Z10 F1000 
      WHILE $AA_IM[X] <= 100; WHILE (Sollwert X-Achse <= 100)
         G1 G91 X10 F500; Bohrbild
         Z–5 F100 
         Z5 
      ENDWHILE 
   ELSE; ELSE Satzsuchlauf
      MSG("Im Suchlauf wird nicht gebohrt") 
   ENDIF; ENDIF
   $A_OUT[1] = 1; nächste Bohrplatte
   G4 F2 
ENDLOOP
