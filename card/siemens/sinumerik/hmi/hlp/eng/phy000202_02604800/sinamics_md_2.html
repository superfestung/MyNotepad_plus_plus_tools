r2122[0], r2124[0], r2123[0], r2125[0] --> alarm 1 (the oldest)

r2122[8], r2124[8], r2123[8], r2125[8] --> Alarm 1 (the latest)
. . .
r2122[63], r2124[63], r2123[63], r2125[63] --> alarm 56 (the oldest)
